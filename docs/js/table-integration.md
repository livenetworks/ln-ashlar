# Table Integration Patterns

This document is a comprehensive guide to implementing tables in `ln-ashlar`. The unified `ln-table` component supports two distinct execution patterns using a single component script:
1. **SSR / Markup Mode (Client-Side In-Memory)**: Best for pre-rendered tables from standard backend templates (like PHP, Rails, Blade) where all rows are loaded at once.
2. **Data-Driven Mode (Virtual Scroll / AJAX)**: Best for handling dynamic templates, selection checkboxes, and loading large datasets (up to 10,000+ items) from remote APIs.

---

## 1. SSR / Markup Mode (Server-Rendered Table)

Used when the backend server prints standard `<tr>` rows directly into the HTML body. `ln-table` hydrates the markup and handles sorting, search, and filters client-side.

### HTML Blueprint
```html
<div data-ln-table id="employee-table">
    <table>
        <thead>
            <tr>
                <th colspan="2">
                    <div class="ln-table__toolbar">
                        <form role="search" onsubmit="return false;">
                            <label data-ln-search="employee-table">
                                <svg class="ln-icon ln-icon--sm" aria-hidden="true"><use href="#ln-search"></use></svg>
                                <input type="search" placeholder="Search employees...">
                            </label>
                        </form>
                    </div>
                </th>
            </tr>
            <tr>
                <th data-ln-table-sort="string">
                    Name
                    <button type="button" class="table-sort" data-ln-table-col-sort aria-label="Sort">
                        <svg class="ln-icon" aria-hidden="true" data-ln-table-col-sort-icon="none"><use href="#ln-arrows-sort"></use></svg>
                        <svg class="ln-icon" aria-hidden="true" data-ln-table-col-sort-icon="asc"><use href="#ln-arrow-up"></use></svg>
                        <svg class="ln-icon" aria-hidden="true" data-ln-table-col-sort-icon="desc"><use href="#ln-arrow-down"></use></svg>
                    </button>
                </th>
                <th data-ln-table-sort="string" data-ln-table-filter-col="dept">
                    Department
                    <button type="button" class="table-filter" aria-label="Filter" data-ln-popover-for="filter-dept">
                        <svg class="ln-icon" aria-hidden="true"><use href="#ln-filter"></use></svg>
                    </button>
                </th>
            </tr>
        </thead>
        <tbody>
            <tr>
                <td>Ana Petrova</td>
                <td>Engineering</td>
            </tr>
            <tr>
                <td>Marko Nikolov</td>
                <td>Design</td>
            </tr>
        </tbody>
    </table>

    <template data-ln-table-empty>
        <article class="ln-table__empty-state">
            <h3>No results found</h3>
            <button type="button" data-ln-table-clear>Clear all</button>
        </article>
    </template>
</div>
```

## 2. Data-Driven Mode (Virtual Scroll & AJAX)

Opted in by specifying the `data-ln-table-source` attribute. Rows are dynamically generated by rendering templates using data payloads.

### HTML Blueprint
```html
<section data-ln-table="products" data-ln-table-source="products" data-ln-table-selectable id="products-table" class="data-table-demo">
    <table>
        <thead>
            <tr>
                <th colspan="3">
                    <div class="ln-table__toolbar">
                        <div>
                            <h3>Products List</h3>
                        </div>
                        <form role="search" onsubmit="return false;">
                            <label>
                                <svg class="ln-icon ln-icon--sm" aria-hidden="true"><use href="#ln-search"></use></svg>
                                <input type="search" placeholder="Search..." data-ln-search="products-table">
                            </label>
                        </form>
                    </div>
                </th>
            </tr>
            <tr>
                <th data-ln-table-col-select></th>
                <th data-ln-table-col="name">
                    Product Name
                    <button type="button" class="table-sort" data-ln-table-col-sort aria-label="Sort">
                        <svg class="ln-icon" aria-hidden="true" data-ln-table-col-sort-icon="none"><use href="#ln-arrows-sort"></use></svg>
                        <svg class="ln-icon" aria-hidden="true" data-ln-table-col-sort-icon="asc"><use href="#ln-arrow-up"></use></svg>
                        <svg class="ln-icon" aria-hidden="true" data-ln-table-col-sort-icon="desc"><use href="#ln-arrow-down"></use></svg>
                    </button>
                </th>
                <th data-ln-table-col="category">
                    Category
                    <button type="button" class="table-filter" data-ln-table-col-filter aria-label="Filter">
                        <svg class="ln-icon" aria-hidden="true"><use href="#ln-filter"></use></svg>
                    </button>
                </th>
            </tr>
        </thead>
        <tbody data-ln-table-body></tbody>
    </table>

    <footer class="ln-table__footer">
        <span data-ln-table-total></span> items
        <span data-ln-table-filtered-wrap> · <span data-ln-table-filtered></span> filtered</span>
        <span data-ln-table-selected-wrap class="hidden"> · <span data-ln-table-selected></span> selected</span>
    </footer>

    <!-- Row Template -->
    <template data-ln-template="products-row">
        <tr data-ln-table-row>
            <td><input type="checkbox" data-ln-table-row-select></td>
            <td>{{ name }}</td>
            <td>{{ category }}</td>
        </tr>
    </template>

    <!-- Empty Filtered State -->
    <template data-ln-template="products-empty-filtered">
        <tr>
            <td colspan="99">
                <article class="ln-table__empty-state">
                    <h3>No results found</h3>
                    <button data-ln-table-clear-all class="btn">Clear Filters</button>
                </article>
            </td>
        </tr>
    </template>

    <!-- Column Filter Template -->
    <template data-ln-template="column-filter">
        <div class="column-filter-dropdown">
            <input type="search" data-ln-filter-search placeholder="Search options...">
            <ul data-ln-filter-options>
                <li>
                    <label>
                        <input type="checkbox" data-ln-filter-reset>
                        All
                    </label>
                </li>
                <template data-ln-template="column-filter-item">
                    <li>
                        <label>
                            <input type="checkbox" data-ln-attr="value:value">
                            <span data-ln-field="value"></span>
                        </label>
                    </li>
                </template>
            </ul>
        </div>
    </template>
</section>
```

### JS Coordinator Example (AJAX Server Fetch)
```javascript
(function () {
    const tableEl = document.getElementById('products-table');
    if (!tableEl) return;

    tableEl.addEventListener('ln-table:request-data', function (e) {
        if (e.detail.table !== 'products') return;

        // 1. Show loading state
        tableEl.dispatchEvent(new CustomEvent('ln-table:set-loading', {
            detail: { loading: true }
        }));

        const params = new URLSearchParams();
        if (e.detail.search) params.append('search', e.detail.search);
        if (e.detail.sort) {
            params.append('sort_field', e.detail.sort.field);
            params.append('sort_dir', e.detail.sort.direction);
        }

        // 2. Fetch dataset
        fetch('/api/products?' + params.toString())
            .then(res => res.json())
            .then(payload => {
                // 3. Dispatch set-data back to table (which automatically clears loading)
                tableEl.dispatchEvent(new CustomEvent('ln-table:set-data', {
                    detail: {
                        data: payload.data,
                        total: payload.total,
                        filtered: payload.filtered,
                        filterOptions: {
                            category: ['Electronics', 'Books', 'Home']
                        }
                    }
                }));
            });
    });
})();
```
