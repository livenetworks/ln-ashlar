    HTML

    <!-- Breadcrumbs -->
<nav class="breadcrumbs" aria-label="Breadcrumbs">
    <ol>
        <li><a href="#">Home</a></li>
        <li><a href="#">Section</a></li>
        <li aria-current="page">Current</li>
    </ol>
</nav>

<!-- Section -->
<section>
    <header>
        <h2>Title</h2>
        <div class="section-actions"><button>Action</button></div>
    </header>
    <main>...</main>
</section>

<!-- Section Card -->
<section class="section-card">
    <header><h3>Title</h3></header>
    <main>...</main>
    <footer><button>Save</button></footer>
</section>

<!-- Inline Alert -->
<div class="alert" role="alert">
    <svg class="ln-icon" aria-hidden="true"><use href="#ln-info-circle"></use></svg>
    <p>Informational message.</p>
    <button type="button" aria-label="Dismiss">...</button>
</div>
<div class="alert success" role="alert">...</div>
<div class="alert warning" role="alert">...</div>
<div class="alert error" role="alert">...</div>

<!-- Banner — full-width variant of alert, dismissed via ln-toggle -->
<div class="alert banner" id="trial-notice" role="status" aria-live="polite"
     data-ln-toggle="open" data-ln-persist>
    <svg class="ln-icon" aria-hidden="true"><use href="#ln-info-circle"></use></svg>
    <p>Trial expires in 3 days.</p>
    <button type="button" aria-label="Dismiss"
            data-ln-toggle-for="trial-notice"
            data-ln-toggle-action="close">
        <svg class="ln-icon" aria-hidden="true"><use href="#ln-x"></use></svg>
    </button>
</div>

<!-- Color variants — same status classes as alert -->

<div class="alert banner warning">...</div>
