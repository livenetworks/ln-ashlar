    HTML

    <!-- Breadcrumbs -->
<nav class="breadcrumbs" aria-label="Breadcrumbs">
    <ol>
        <li><a href="#">Home</a></li>
        <li><a href="#">Section</a></li>
        <li aria-current="page">Current</li>
    </ol>
</nav>

<!-- Section -->
<section>
    <header>
        <h2>Title</h2>
        <div class="section-actions"><button>Action</button></div>
    </header>
    <main>...</main>
</section>

<!-- Section Card -->
<section class="section-card">
    <header><h3>Title</h3></header>
    <main>...</main>
    <footer><button>Save</button></footer>
</section>

<!-- Inline Alert -->
<div class="alert" role="alert">
    <svg class="ln-icon" aria-hidden="true"><use href="#ln-info-circle"></use></svg>
    <p>Informational message.</p>
    <button type="button" aria-label="Dismiss">...</button>
</div>
<div class="alert success" role="alert">...</div>
<div class="alert warning" role="alert">...</div>
<div class="alert error" role="alert">...</div>

<!-- Banner (dismissible, inline, uses ln-modal JS) -->
<div class="ln-modal ln-banner" id="my-banner" data-ln-modal="open">
    <form>
        <svg class="ln-icon" aria-hidden="true"><use href="#ln-info-circle"></use></svg>
        <p>Banner message.</p>
        <button type="button" aria-label="Dismiss" data-ln-modal-close></button>
    </form>
</div>

<!-- Color variants via CSS variable -->
<div class="ln-modal ln-banner success" data-ln-modal="open">...</div>
<div class="ln-modal ln-banner warning" data-ln-modal="open">...</div>
<div class="ln-modal ln-banner error" data-ln-modal="open">...</div>

<!-- Trigger to reopen -->
<button type="button" data-ln-modal-for="my-banner">Show banner</button>
