    Sortable

    Drag & drop reorder component using the Pointer Events API. Works with touch + mouse. The component only reorganizes the DOM — data model sync is the consumer's responsibility via events.

    Attributes

    AttributeOn elementDescription
    data-ln-sortablecontainerCreates an instance. Sortable items = direct children.
    data-ln-sortable-handleelement inside a childDrag handle. If absent — the entire child is draggable.

    Simple List

    No handle — drag anywhere on the item.

    Drag only by the handle.

      CSS Classes

      The component adds/removes these classes. Does not ship CSS — the consumer styles them.

      ClassOnWhen
      .ln-sortable--activecontainerWhile drag is in progress
      .ln-sortable--draggingdragged elementWhile being dragged
      .ln-sortable--drop-beforetarget elementPointer in top half
      .ln-sortable--drop-aftertarget elementPointer in bottom half

      Events

      EventBubblesCancelabledetail
      ln-sortable:before-dragyesyes{ item, index }
      ln-sortable:drag-startyesno{ item, index }
      ln-sortable:reorderedyesno{ item, oldIndex, newIndex }
      ln-sortable:enabledyesno{ target }
      ln-sortable:disabledyesno{ target }
       
      document.addEventListener('ln-sortable:reordered', function (e) {
    console.log('Moved from', e.detail.oldIndex, 'to', e.detail.newIndex);
});

// Cancel drag conditionally
document.addEventListener('ln-sortable:before-drag', function (e) {
    if (listIsLocked) e.preventDefault();
});

// Disable via attribute (single source of truth)
list.setAttribute('data-ln-sortable', 'disabled');

      HTML Structure

      <!-- Without handle (entire item draggable) -->
<ul data-ln-sortable>
    <li>Drag me</li>
    <li>Me too</li>
</ul>

<!-- With handle -->
<ol data-ln-sortable>
    <li>
        <span data-ln-sortable-handle>⋮⋮</span>
        First item
    </li>
    <li>
        <span data-ln-sortable-handle>⋮⋮</span>
        Second item
    </li>
</ol>

<!-- Required CSS -->
<style>
[data-ln-sortable-handle] {
    touch-action: none;
    cursor: grab;
}
.my-list > li.ln-sortable--dragging { opacity: 0.4; }
.my-list > li.ln-sortable--drop-before { box-shadow: inset 0 2px 0 0 var(--color-primary); }
.my-list > li.ln-sortable--drop-after  { box-shadow: inset 0 -2px 0 0 var(--color-primary); }
</style>

      API

      var list = document.querySelector('[data-ln-sortable]');

// Via API (sets the attribute, observer syncs state)
list.lnSortable.enable();     // sets data-ln-sortable=""
list.lnSortable.disable();    // sets data-ln-sortable="disabled"
list.lnSortable.isEnabled;    // boolean

// Via attribute (identical result)
list.setAttribute('data-ln-sortable', 'disabled');
list.setAttribute('data-ln-sortable', '');

// Constructor — only for non-standard cases
window.lnSortable(container);
