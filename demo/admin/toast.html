    Toast

    Notifications with a side accent and icon. They automatically disappear after 6 seconds.

    Examples

    Error with a list of errors

    HTML & JS API

    <!-- Toast container (once per page, no template needed) -->
<ul data-ln-toast></ul>

<!-- JS API -->
<script>
// Basic toast — returns numeric id
var id = window.lnToast.enqueue({
    type: 'success',     // success | error | warn | info
    title: 'Title',
    message: 'Description...'
});

// Error with list
window.lnToast.enqueue({
    type: 'error',
    title: 'Validation',
    message: ['Error 1', 'Error 2', 'Error 3']
});

// Clear all
window.lnToast.clear();
</script>

<!-- Static toast (hydrated on init) -->
<ul data-ln-toast>
    <li data-ln-toast-item data-type="success" data-title="OK">
        Saved!
    </li>
</ul>
