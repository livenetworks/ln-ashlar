    Autosave

    Auto-saves form data to localStorage on field blur/change. Restores on next visit. Clears on submit, reset, or cancel.

    Basics

    AttributeOn elementDescription
    data-ln-autosave<form>Enables autosave. Uses form.id as storage key.
    data-ln-autosave="key"<form>Explicit storage key (overrides form.id).
    data-ln-autosave-clearbutton inside formClick clears saved draft (cancel/discard buttons).

    Storage key format: ln-autosave:{pathname}:{identifier} — unique per URL + form.

    Live demo — fill fields, close tab, reopen

    Events

    EventCancelableDetailWhen
    ln-autosave:before-restoreyes{ target, data }Before restoring saved data
    ln-autosave:restoredno{ target, data }Data restored to form
    ln-autosave:savedno{ target, data }Data saved to localStorage
    ln-autosave:clearedno{ target }localStorage entry removed
    ln-autosave:destroyedno{ target }Instance destroyed

    API
    
						
    
							
    // Instance API (on DOM element)
var form = document.getElementById('my-form');
form.lnAutosave.save();     // manually save
form.lnAutosave.restore();  // manually restore
form.lnAutosave.clear();    // remove saved data
form.lnAutosave.destroy();  // cleanup listeners

// Constructor (for dynamic content)
window.lnAutosave(container);
    
						
    
					
    

					
					
    
						
    Cancel restore (server data takes priority)
    
						
    
							
    // Prevent restore when form already has server data
document.addEventListener('ln-autosave:before-restore', function (e) {
    var form = e.detail.target;
    if (form.dataset.hasServerData === 'true') {
        e.preventDefault();       // don't overwrite server values
        form.lnAutosave.clear();  // discard stale draft
    }
});
