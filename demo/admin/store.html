    Store

    Generic IndexedDB data layer — caches server data locally, syncs via delta protocol, handles optimistic mutations. Renders nothing — UI components consume data through its API and CustomEvents.

    Attributes

    AttributeRequiredDescription
    data-ln-storeyesStore name (IndexedDB object store name, event namespace)
    data-ln-store-endpointyesServer API base URL for this resource
    data-ln-store-stalenoStaleness threshold in seconds (default: 300). "0" = always stale (syncs on every mount). "-1" or "never" = never auto-sync (only forceSync())
    data-ln-store-indexesnoComma-separated fields to index
    data-ln-store-search-fieldsnoComma-separated fields for text search

      Events — Commands (dispatched TO the store)
      
						
      
							
      
							
      
								
      EventdetailDescription
      
								
									
      ln-store:request-create{ data }Create record (optimistic with temp ID)
      
									
      ln-store:request-update{ id, data, expected_version }Update record (optimistic, conflict detection)
      
									
      ln-store:request-delete{ id }Delete record (optimistic)
      
									
      ln-store:request-bulk-delete{ ids }Delete multiple records (optimistic)
      
								      
							
      
							
      
						
      
					
      

					
					
      
						
      Events — Notifications (emitted BY the store)
      
						
      
							
      Data Events
      
							
      
							
      
								
      EventWhendetail
      
								
									
      ln-store:readyData available{ store, count, source }
      
									
      ln-store:loadedInitial full load{ store, count }
      
									
      ln-store:syncedDelta sync complete{ store, added, deleted, changed: bool }
      
								      
							
      
							
      

							
      Mutation Events
      
							
      
							
      
								
      EventWhendetail
      
								
									
      ln-store:createdOptimistic create{ store, record, tempId }
      
									
      ln-store:updatedOptimistic update{ store, record, previous }
      
									
      ln-store:deletedOptimistic delete{ store, id } or { store, ids }
      
									
      ln-store:confirmedServer confirmed{ store, record, action }
      
									
      ln-store:revertedMutation failed{ store, record, action, error }
      
									
      ln-store:conflictUpdate conflict (409){ store, local, remote, field_diffs }
      
								      
							
      
							
      

							
      Error Events
      
							
      
							
      
								
      EventWhendetail
      
								
									
      ln-store:errorInitial load failed{ store, action, error, status }
      
									
      ln-store:offlineServer unreachable{ store }
      
									
      ln-store:destroyedInstance destroyed{ store }
      
								      
							
      
							
      
						
      
					
      

					
					
      
						
      Sync Lifecycle
      
						
      
							
      Init:
  1. Open IndexedDB → read _meta
  2. Schema mismatch? → clear, full load
  3. Has cache? → emit ready (source: 'cache'), delta sync if stale
  4. No cache? → full load → emit loaded, ready (source: 'server')

Visibility change (tab returns):
  → delta sync all stores if stale

Delta sync:
  1. GET {endpoint}?since={last_synced_at}
  2. Upsert data[], delete deleted[]
  3. Emit synced
      
						
      
					
      

					
					
      
						
      Optimistic Mutation Flow
      
						
      
							
      Create:  insert temp → POST → confirmed | reverted
Update:  save prev, update → PUT → confirmed | conflict | reverted
Delete:  save, remove → DELETE → confirmed | reverted
Bulk:    save all, remove → DELETE /bulk-delete → confirmed | reverted
      
						
      
					
      

					
					
      
						
      API
      
						
      
							
      const storeEl = document.querySelector('[data-ln-store="documents"]');

// Queries (all return Promise)
storeEl.lnStore.getAll(options)        // → { data, total, filtered }
storeEl.lnStore.getById(id)            // → Object|null
storeEl.lnStore.count(filters)         // → Number
storeEl.lnStore.aggregate(field, fn)   // → Number (sum, avg, count)

// State
storeEl.lnStore.isLoaded               // Boolean
storeEl.lnStore.isSyncing              // Boolean
storeEl.lnStore.lastSyncedAt           // Number|null
storeEl.lnStore.totalCount             // Number

// Manual triggers
storeEl.lnStore.forceSync()            // delta sync
storeEl.lnStore.fullReload()           // clear + full reload
storeEl.lnStore.destroy()              // cleanup

// Global
window.lnStore.clearAll()              // wipe all stores (logout)
window.lnStore.init(el)                // manual init
      
						
      
					
      

					
					
      
						
      Query Options
      
						
      
							
      storeEl.lnStore.getAll({
    sort: { field: 'title', direction: 'asc' },
    filters: { status: ['approved', 'draft'] },
    search: 'ISO 27001',
    offset: 0,
    limit: 100
})
// → { data: [...], total: 1247, filtered: 45 }
      
						
      
					
      

					
					
      
						
      Server Response Format
      
						
      
							
      // Full load / Delta sync
{
    "data": [{ "id": 42, "title": "...", "updated_at": 1736952600 }],
    "deleted": [17, 23],          // delta only
    "synced_at": 1736953500       // next ?since= value
}
      
						
      
					
      

					
					
      
						
      HTML Structure
      
						
      
							
      <!-- One per resource, anywhere in the page -->
<div data-ln-store="documents"
     data-ln-store-endpoint="/api/documents"
     data-ln-store-stale="300"
     data-ln-store-indexes="status,category,updated_at"
     data-ln-store-search-fields="title,description,author_name">
</div>
