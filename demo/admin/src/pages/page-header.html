SCSS Usage

// Library default — applies @include page-header to .page-header.
// Requires a container-query context on an ancestor.
//
//   <main class="main">                           ← container-type: inline-size
//       <header class="page-header">
//           <nav aria-label="Breadcrumb">...</nav>
//           <div><h1>...</h1></div>
//           <div><button>...</button></div>
//       </header>
//   </main>

// Project layout establishes the container context
.main { @include container; }  // container-type: inline-size

// Custom project selector (second header, section header, etc.)
#document-detail > header { @include page-header; }
