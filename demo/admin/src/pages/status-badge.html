Production Usage Pattern

In production, avoid class utilities like .success on project-owned elements. Use @include badge on a semantic selector and override --color-primary:

<!-- Library classes (prototyping / design system demos) -->
<ul class="demo-badge-list">
	<li class="success">Active</li>
	<li class="warning">Pending</li>
	<li class="neutral">Inactive</li>
</ul>

<!-- Production pattern — project-owned selectors -->
<ul class="status-list">
	<li class="active">Active</li>
	<li class="pending">Pending</li>
	<li class="inactive">Inactive</li>
</ul>
// Project SCSS
.status-list li {
	@include badge;
	&.active   { --color-primary: var(--color-success); }
	&.pending  { --color-primary: var(--color-warning); }
	&.inactive { --color-primary: var(--color-neutral-400); }
}

// Add pulse for real-time states
.status-list li.live { @include badge-live; }
