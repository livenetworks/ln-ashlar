Sortable

Drag & drop reorder component using the Pointer Events API. Works with touch + mouse. The component only reorganizes the DOM — data model sync is the consumer's responsibility via events.

Attributes

AttributeOn elementDescription
data-ln-sortablecontainerCreates an instance. Sortable items = direct children.
data-ln-sortable-handleelement inside a childDrag handle. If absent — the entire child is draggable.

// Cancel drag conditionally
document.addEventListener('ln-sortable:before-drag', function (e) {
    if (listIsLocked) e.preventDefault();
});

// Disable via attribute (single source of truth)
list.setAttribute('data-ln-sortable', 'disabled');

    API

    var list = document.querySelector('[data-ln-sortable]');

// Via API (sets the attribute, observer syncs state)
list.lnSortable.enable();     // sets data-ln-sortable=""
list.lnSortable.disable();    // sets data-ln-sortable="disabled"
list.lnSortable.isEnabled;    // boolean

// Via attribute (identical result)
list.setAttribute('data-ln-sortable', 'disabled');
list.setAttribute('data-ln-sortable', '');

// Constructor — only for non-standard cases
window.lnSortable(container);
