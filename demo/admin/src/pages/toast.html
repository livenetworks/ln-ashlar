Toast

Notifications with a side accent and icon. They automatically disappear after 6 seconds.

Examples

Error with a list of errors

HTML & JS API

<!-- Template (required once per page) -->
<template data-ln-template="ln-toast-item">
    <li class="ln-toast__item">
        <div class="ln-toast__card" data-ln-attr="role:role, aria-live:ariaLive">
            <div class="ln-toast__side"></div>
            <div class="ln-toast__content">
                <div class="ln-toast__head">
                    <strong class="ln-toast__title" data-ln-field="title"></strong>
                </div>
                <button type="button" class="ln-toast__close" aria-label="Close"><svg class="ln-icon" aria-hidden="true"><use href="#ln-x"></use></svg></button>
                <div class="ln-toast__body" data-ln-show="hasBody"></div>
            </div>
        </div>
    </li>
</template>

<!-- Toast container (once per page) -->
<ul data-ln-toast></ul>

<!-- JS — dispatch event to show a toast -->
<script>
// Basic toast
window.dispatchEvent(new CustomEvent('ln-toast:enqueue', {
    detail: {
        type: 'success',     // success | error | warn | info
        title: 'Title',
        message: 'Description...'
    }
}));

// Error with list
window.dispatchEvent(new CustomEvent('ln-toast:enqueue', {
    detail: {
        type: 'error',
        title: 'Validation',
        message: ['Error 1', 'Error 2', 'Error 3']
    }
}));

// Clear all
window.dispatchEvent(new CustomEvent('ln-toast:clear'));
</script>

<!-- Static toast (hydrated on init) -->
<ul data-ln-toast>
    <li data-ln-toast-item data-type="success" data-title="OK">
        Saved!
    </li>
</ul>
