HTML

<!-- Table: all rows clickable -->
<table data-ln-link>
    <tbody>
        <tr>
            <td><a href="/users/1">Marko</a></td>
            <td>marko@example.com</td>
        </tr>
    </tbody>
</table>

<!-- Generic container -->
<article data-ln-link>
    <h3><a href="/post/1">Title</a></h3>
    <p>Description</p>
</article>

ActionResult

API

// Manual initialization of a new DOM element
window.lnLink.init(document.getElementById('new-container'));

Compatibility with ln-ajax

ln-link and ln-ajax are fully decoupled — they communicate via native DOM click events:

  1. ln-link triggers .click() on <a>
  2. ln-ajax intercepts the click event (if present)
  3. If ln-ajax is not present — normal navigation
<!-- Both components together -->
<section data-ln-ajax>
    <table data-ln-link>
        <tbody>
            <tr>
                <td><a href="/users/1">Marko</a></td>
                <td>marko@example.com</td>
            </tr>
        </tbody>
    </table>
</section>
