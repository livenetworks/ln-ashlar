    Modal

    JS component for modal windows. The data-ln-modal attribute is the single source of truth for open/closed state. ESC closes.

    Code

    // Cancel opening conditionally (cancelable)
document.addEventListener('ln-modal:before-open', function(e) {
    if (e.detail.modalId === 'confirm-dialog' && !formIsValid()) {
        e.preventDefault(); // the modal will not open
    }
});

// Prevent closing (unsaved changes)
document.getElementById('edit-modal').addEventListener('ln-modal:before-close', function(e) {
    if (hasUnsavedChanges()) {
        e.preventDefault();
    }
});

document.addEventListener('ln-modal:open', function(e) {
    console.log('Opened:', e.detail.modalId);
});

document.addEventListener('ln-modal:close', function(e) {
    if (e.detail.modalId === 'confirm-dialog') {
        // reset form, clear state etc.
    }
});

    Opening and closing this modal dispatches ln-modal:open and ln-modal:close events.

    HTML

    <!-- Trigger button -->
<button data-ln-modal-for="my-modal">Open</button>

<!-- Modal — form is always the content root -->
<div class="ln-modal" data-ln-modal id="my-modal">
    <form>
        <header>
            <h3>Title</h3>
            <button type="button" data-ln-modal-close aria-label="Close"><svg class="ln-icon" aria-hidden="true"><use href="#ln-x"></use></svg></button>
        </header>
        <main>
            <div class="form-element">
                <label for="name">Name</label>
                <input type="text" id="name">
            </div>
        </main>
        <footer>
            <button type="button" data-ln-modal-close>Cancel</button>
            <button type="submit">Save</button>
        </footer>
    </form>
</div>

<!-- JS API (sets the attribute, observer does the rest) -->
<script>
    const modal = document.getElementById('my-modal');
    modal.lnModal.open();       // sets data-ln-modal="open"
    modal.lnModal.close();      // sets data-ln-modal="close"

    // Direct attribute change — identical result
    modal.setAttribute('data-ln-modal', 'open');
    modal.setAttribute('data-ln-modal', 'close');
</script>
