    Upload

    File upload with drag-drop, progress bar, file type validation. A server is required for actual upload.

    Upload Zone (demo — no server)

    Drag files here or click to browse

    Allowed: pdf, doc, docx, jpg, png

      Custom Item Template (scoped override)

      This uploader provides its own <template> inside the container — the default is ignored for this instance only.

      Drag files here — two-line layout

        HTML

        <!-- Upload container -->
<div data-ln-upload="/api/upload-endpoint"
     data-ln-upload-accept="pdf,doc,jpg,png">

    <!-- Drop zone -->
    <div class="ln-upload__zone">
        <p>Drag files here</p>
    </div>

    <!-- File list (populated automatically) -->
    <ul class="ln-upload__list"></ul>

</div>

<!-- Attributes -->
data-ln-upload="URL"           - upload endpoint
data-ln-upload-accept="ext,ext" - allowed extensions

<!-- JS API (available on the container element) -->
<script>
const container = document.querySelector('[data-ln-upload]');
container.lnUploadAPI.getFileIds();   // [1, 2, 3]
container.lnUploadAPI.getFiles();     // file objects
container.lnUploadAPI.clear();        // clear everything
</script>

<!-- Custom Events -->
ln-upload:uploaded   - {id, name, size}
ln-upload:removed    - {id}
ln-upload:error      - {message}
ln-upload:invalid    - {file, reason}

<!-- Hidden inputs for form -->
<!-- Created automatically: -->
<input type="hidden" name="file_ids[]" value="123">

<!-- Custom item template (optional scoped override) -->
<div data-ln-upload="/api/upload">
    <template data-ln-template="ln-upload-item">
        <li class="ln-upload__item"
            data-ln-class="ln-upload__item--uploading:uploading, ln-upload__item--error:error, ln-upload__item--deleting:deleting">
            <svg class="ln-icon" aria-hidden="true"><use data-ln-attr="href:iconHref" href="#ln-file"></use></svg>
            <span data-ln-field="name"></span>
            <span data-ln-field="sizeText"></span>
            <button type="button" data-ln-upload-action="remove" data-ln-attr="aria-label:removeLabel, title:removeLabel">...</button>
            <div class="ln-upload__progress"><div class="ln-upload__progress-bar"></div></div>
        </li>
    </template>
    <div class="ln-upload__zone"><p>Drop files</p></div>
    <ul class="ln-upload__list"></ul>
</div>
