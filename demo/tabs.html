Code

document.getElementById('user-tabs').addEventListener('ln-tabs:change', function(e) {
    console.log('Active tab:', e.detail.key);
});

// Global — for all tab sections
document.addEventListener('ln-tabs:change', function(e) {
    analytics.track('tab_change', { tab: e.detail.key });
});

HTML

<!-- Namespace = id attribute (or data-ln-tabs-key) -->
<section id="my-tabs" data-ln-tabs data-ln-tabs-default="first">

    <!-- Tab buttons -->
    <nav>
        <button data-ln-tab="first">First tab</button>
        <button data-ln-tab="second">Second tab</button>
    </nav>

    <!-- Panels -->
    <section data-ln-panel="first">
        Content of the first tab
    </section>

    <section data-ln-panel="second" class="hidden">
        Content of the second tab
    </section>

</section>

<!-- Attributes -->
id="key"                   - namespace for hash (required for multiple sections)
data-ln-tabs-key="key"     - alternative, if id is not suitable
data-ln-tabs               - container
data-ln-tabs-default="x"   - default tab
data-ln-tabs-focus         - focus first input on change
data-ln-tab="key"          - tab button
data-ln-panel="key"        - panel content

<!-- Hash format: #my-tabs:second → activate second tab -->
<!-- Multiple sections: #my-tabs:second&other-tabs:info -->
<!-- Without id/data-ln-tabs-key → no hash sync, only defaultKey -->
